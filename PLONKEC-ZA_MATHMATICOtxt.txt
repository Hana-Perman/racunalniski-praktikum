MATRIKE, LASTNI VEKTORJI, LASTNE VREDNOSTI
M = {{9, 2, -3}, {3, 2, 1}, {6, 4, -1}}
Det[M]
vecs = Eigenvectors[M]  ->to je lastni vektor
vals = Eigenvalues[M]	->to so lastne vrednosti
DiagonalizableMatrixQ[M] ->ali je diagonazibilna
R = DiagonalMatrix[vals] ->diagonalna matrika
P = Transpose[vecs] ->transponiranka
Q = Inverse[P] ->inverzna matrika
Simplify[M == P . R . Q]
